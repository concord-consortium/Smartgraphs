Commands related to graphs, tables, annotations

  ACTIVITY

    createLineThroughPoints => LineThroughPoints
      lineName
      firstPoint
      secondPoint
      color (default black)

    createRiseArrow => Arrow
      arrowName
      firstPoint
      secondPoint
      color (default black)
      (note label is 'Rise' and labelUnitKey is 'yLabelAbbreviated')
    
    createRunArrow => Arrow
      arrowName
      firstPoint
      secondPoint
      color (default black)
      (note label is 'Run' and labelUnitKey is 'xLabelAbbreviated')
    
    createIndicatingArrowFromDataPoint => IndicatingArrow
      (** requires a way to specify data point)
      (** graphName NOT required)
      arrowName
      point
      angle
      color (** NO default)
      (** no 'length' argument)
    
    createIndicatingArrowFromHighlightedPoint => IndicatingArrow
      arrowName
      point (is name of highlighted point)
      angle (default 335)
      color (default "cc0000" -- reddish/pinkish)
      length (default 40)
    
    createIndicatingArrowFromCoordinates => IndicatingArrow
      arrowName
      x
      y
      angle (default 335)
      color (default "cc0000" -- reddish/pinkish)
      length (default 40)
    
    createBracketArcFromCoordinates => BracketArc
      bracketName
      startX
      startY
      endX
      endY
      color (default "cc0000" -- reddish/pinkish)    
      
    createRiseBracket => BracketArc
      datasetName
      point1
      point2
      color (default "cc0000" -- reddish/pinkish)
      ** NOTE this creates a BracketArc with hardwired coordinates
      
    createRunBracket => BracketArc
      datasetName
      point1
      point2
      color (default "cc0000" -- reddish/pinkish)
      ** NOTE this creates a BracketArc with hardwired coordinates
      
    toggleAnnotationHighlight
      graphName
      annotationName
